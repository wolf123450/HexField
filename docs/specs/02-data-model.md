## 1. Core TypeScript Types — `src/types/core.ts`

```typescript
// ── Identity ──────────────────────────────────────────────────────────────────
export interface Identity {
  userId:          string        // UUID v7, generated locally on first launch
  displayName:     string
  avatarUrl?:      string        // data: URI or remote URL
  publicSignKey:   string        // Ed25519 public key, base64
  publicDHKey:     string        // X25519 public key, base64
  // Private keys are NEVER stored in this type — see cryptoService
}

// ── Server (community / guild) ─────────────────────────────────────────────────
export interface Server {
  id:              string
  name:            string
  description?:    string
  iconUrl?:        string
  ownerId:         string
  memberCount:     number
  createdAt:       string        // ISO 8601
  inviteCode?:     string
  customEmoji:     CustomEmoji[]
  historyStartsAt?: string       // set when server is re-baselined (Phase 6)
}

export interface ServerMember {
  userId:          string
  serverId:        string
  displayName:     string        // server-specific nickname
  roles:           string[]
  joinedAt:        string
  publicSignKey:   string        // cached for signature verification
  publicDHKey:     string        // cached for encryption
  onlineStatus:    OnlineStatus
}

export type OnlineStatus = 'online' | 'idle' | 'dnd' | 'offline'

// ── Channels ───────────────────────────────────────────────────────────────────
export type ChannelType = 'text' | 'voice' | 'announcement'

export interface Channel {
  id:              string
  serverId:        string
  name:            string
  type:            ChannelType
  position:        number
  topic?:          string
  lastMessageId?:  string
  lastMessageAt?:  string
}

// ── Messages ───────────────────────────────────────────────────────────────────
export interface Message {
  id:              string        // UUID v7 (time-sortable), generated by sender
  channelId:       string
  serverId:        string
  authorId:        string
  content:         string | null // null when deleted — content erased locally, record kept for sync
  contentType:     'text' | 'markdown' | 'system'
  attachments:     Attachment[]  // empty after deletion (files purged from disk)
  reactions:       ReactionSummary[]  // computed from mutations log
  isEdited:        boolean       // true if any 'edit' mutation exists for this message id
  replyToId?:      string
  logicalTs:       string        // Hybrid Logical Clock — canonical ordering
  createdAt:       string        // wall clock — display only
  verified:        boolean       // Ed25519 signature was verified
}

// A mutation is its own syncable event (own UUID v7, own Negentropy entry)
export type MutationType =
  | 'edit' | 'delete' | 'reaction_add' | 'reaction_remove'  // message-level
  | 'server_update'                                          // server name/icon/description
  | 'role_assign' | 'role_revoke'                           // member role changes
  | 'channel_create' | 'channel_update' | 'channel_delete'  // channel management
  | 'device_attest' | 'device_revoke'                       // device linking

export interface Mutation {
  id:              string        // UUID v7
  type:            MutationType
  targetId:        string        // message id, server id, channel id, or device id
  channelId:       string        // '__server__' for server/role/device mutations
  authorId:        string
  newContent?:     string        // 'edit' and 'server_update' only (JSON string for server_update)
  emojiId?:        string        // 'reaction_add' | 'reaction_remove' only
  logicalTs:       string
  createdAt:       string
  verified:        boolean
}

export interface Attachment {
  id:              string
  name:            string
  size:            number
  mimeType:        string
  url?:            string        // blob: after download; omit for Phase 1 inline
  inlineData?:     string        // base64 for Phase 1 ≤100KB inline attachments
  contentHash?:    string        // blake3: for Phase 5b P2P attachments
  transferState:   'pending' | 'transferring' | 'complete' | 'failed'
}

// ── Reactions ──────────────────────────────────────────────────────────────────
export interface ReactionSummary {
  emojiId:         string        // Unicode codepoint string or custom emoji UUID
  count:           number
  selfReacted:     boolean
}

export interface CustomEmoji {
  id:              string
  serverId:        string
  name:            string        // shortcode without colons
  uploadedBy:      string
  createdAt:       string
  // image is NOT stored here — loaded lazily via get_emoji_image Rust command
}

// ── Encryption envelope ────────────────────────────────────────────────────────
export interface EncryptedEnvelope {
  version:         1
  senderId:        string
  recipientId:     string
  ciphertext:      string        // base64 XSalsa20-Poly1305 output
  nonce:           string        // base64 24-byte nonce
  senderSignature: string        // base64 Ed25519 signature over (ciphertext + nonce)
}

// ── Peer / WebRTC ──────────────────────────────────────────────────────────────
export interface Peer {
  userId:          string
  connectionState: RTCPeerConnectionState
  audioEnabled:    boolean
  videoEnabled:    boolean
  screenSharing:   boolean
  speaking:        boolean       // VAD result
}

export interface VoiceSession {
  channelId:       string
  serverId:        string
  joinedAt:        string
  peers:           Record<string, Peer>
}

export interface InviteLink {
  code:            string
  serverId:        string
  createdBy:       string
  expiresAt?:      string
  maxUses?:        number
  uses:            number
}

// ── Device ─────────────────────────────────────────────────────────────────────
export interface Device {
  deviceId:        string
  userId:          string
  publicSignKey:   string
  publicDHKey:     string
  attestedBy?:     string        // deviceId of attesting device (null for root)
  attestationSig?: string        // base64 Ed25519 signature from attesting device
  revoked:         boolean
  createdAt:       string
}
```

## 2. SQLite Schema — `src-tauri/migrations/001_initial.sql`

```sql
CREATE TABLE IF NOT EXISTS servers (
  id           TEXT PRIMARY KEY,
  name         TEXT NOT NULL,
  description  TEXT,
  icon_url     TEXT,
  owner_id     TEXT NOT NULL,
  invite_code  TEXT,
  created_at   TEXT NOT NULL,
  raw_json     TEXT NOT NULL      -- full serialized Server object
);

CREATE TABLE IF NOT EXISTS channels (
  id           TEXT PRIMARY KEY,
  server_id    TEXT NOT NULL REFERENCES servers(id) ON DELETE CASCADE,
  name         TEXT NOT NULL,
  type         TEXT NOT NULL DEFAULT 'text',
  position     INTEGER NOT NULL DEFAULT 0,
  topic        TEXT,
  created_at   TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS messages (
  id              TEXT PRIMARY KEY,        -- UUID v7 (time-sortable)
  channel_id      TEXT NOT NULL REFERENCES channels(id) ON DELETE CASCADE,
  server_id       TEXT NOT NULL,
  author_id       TEXT NOT NULL,
  content         TEXT,                    -- NULL when deleted (content erased, record kept for sync)
  content_type    TEXT NOT NULL DEFAULT 'text',
  reply_to_id     TEXT,
  created_at      TEXT NOT NULL,
  logical_ts      TEXT NOT NULL,           -- Hybrid Logical Clock value for canonical ordering
  verified        INTEGER NOT NULL DEFAULT 0,
  raw_attachments TEXT                     -- JSON array; attachment files deleted separately on delete
);
CREATE INDEX IF NOT EXISTS idx_messages_channel ON messages(channel_id, logical_ts DESC);
-- Full-text search (Phase 6) — only indexes non-null content:
CREATE VIRTUAL TABLE IF NOT EXISTS messages_fts USING fts5(
  content, content=messages, content_rowid=rowid
);

-- Mutations log: edits, deletes, reactions, and server-level events as syncable events.
-- channel_id = '__server__' for server/role/device/channel mutations.
-- Reconciled via second (per-channel) and third (per-server) Negentropy passes.
CREATE TABLE IF NOT EXISTS mutations (
  id              TEXT PRIMARY KEY,        -- UUID v7 (own syncable identity)
  type            TEXT NOT NULL,           -- see MutationType
  target_id       TEXT NOT NULL,           -- message/server/channel/device id
  channel_id      TEXT NOT NULL,           -- denormalized; '__server__' for server-level events
  author_id       TEXT NOT NULL,
  new_content     TEXT,                    -- 'edit' payload or JSON for 'server_update'
  emoji_id        TEXT,                    -- reaction types only
  logical_ts      TEXT NOT NULL,
  created_at      TEXT NOT NULL,
  verified        INTEGER NOT NULL DEFAULT 0
);
CREATE INDEX IF NOT EXISTS idx_mutations_channel ON mutations(channel_id, logical_ts DESC);
CREATE INDEX IF NOT EXISTS idx_mutations_target  ON mutations(target_id);

CREATE TABLE IF NOT EXISTS members (
  user_id          TEXT NOT NULL,
  server_id        TEXT NOT NULL REFERENCES servers(id) ON DELETE CASCADE,
  display_name     TEXT NOT NULL,
  roles            TEXT,                   -- JSON array
  joined_at        TEXT NOT NULL,
  public_sign_key  TEXT NOT NULL,
  public_dh_key    TEXT NOT NULL,
  online_status    TEXT NOT NULL DEFAULT 'offline',
  PRIMARY KEY (user_id, server_id)
);

CREATE TABLE IF NOT EXISTS custom_emoji (
  id           TEXT PRIMARY KEY,
  server_id    TEXT NOT NULL REFERENCES servers(id) ON DELETE CASCADE,
  name         TEXT NOT NULL,
  file_path    TEXT NOT NULL,              -- relative to $APPDATA/gamechat/emoji/{serverId}/{id}.webp
  uploaded_by  TEXT NOT NULL,
  created_at   TEXT NOT NULL
  -- image bytes stored on disk, NOT in SQLite
);

CREATE TABLE IF NOT EXISTS key_store (
  key_id       TEXT PRIMARY KEY,
  key_type     TEXT NOT NULL,              -- 'sign_secret' | 'dh_secret'
  key_data     TEXT NOT NULL,              -- base64 (plain Phase 1, passphrase-wrapped Phase 2)
  created_at   TEXT NOT NULL
);

-- Device attestations for multi-device support (Phase 4b)
CREATE TABLE IF NOT EXISTS devices (
  device_id        TEXT PRIMARY KEY,
  user_id          TEXT NOT NULL,
  public_sign_key  TEXT NOT NULL,
  public_dh_key    TEXT NOT NULL,
  attested_by      TEXT,                   -- device_id of attesting device (NULL for root device)
  attestation_sig  TEXT,                   -- Ed25519 signature from attesting device
  revoked          INTEGER NOT NULL DEFAULT 0,
  created_at       TEXT NOT NULL
);
CREATE INDEX IF NOT EXISTS idx_devices_user ON devices(user_id);

-- Per-channel permission overrides (see spec 10-permissions.md)
CREATE TABLE IF NOT EXISTS channel_permission_overrides (
  channel_id   TEXT NOT NULL REFERENCES channels(id) ON DELETE CASCADE,
  role_name    TEXT NOT NULL,
  permission   TEXT NOT NULL,              -- permission flag name
  effect       TEXT NOT NULL,              -- 'allow' | 'deny'
  PRIMARY KEY (channel_id, role_name, permission)
);
```

## 3. File System Layout

```
$APPDATA/gamechat/
  gamechat.db                    ← SQLite database
  emoji/
    {serverId}/
      {emojiId}.webp             ← 128×128 WebP, 5–15KB each
  attachments/                   ← Phase 5b only
    {hash[0:2]}/
      {hash}.bin                 ← complete file
      {hash}.part                ← incomplete download
```

## 4. Message ID Conventions

| ID Field | Format | Rationale |
|----------|--------|-----------|
| `userId` | UUID v7 | Consistent with other IDs; time component unused but harmless |
| `messageId` | UUID v7 | Time-sortable; simplifies Negentropy range partitioning |
| `mutationId` | UUID v7 | Same — mutations are first-class syncable events |
| `deviceId` | UUID v7 | Consistent |
| `serverId`, `channelId`, `emojiId` | UUID v7 | Consistent |

All IDs generated client-side. No server assignment needed.
